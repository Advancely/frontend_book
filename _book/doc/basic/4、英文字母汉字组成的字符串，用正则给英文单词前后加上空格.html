let str = "test一个测试，just测试一下下smileyqp",
        reg = /\b[a-z]+\b/ig;        //后面的g是全局匹配，i是忽略大小写
str.replace(reg,value=>{  //value正则捕获的内容
    return ' '+value+' '
}).trim();                            //trim()去除开头和结尾的空格；trimLeft()去除开头空格；trimRight()去结尾空格
